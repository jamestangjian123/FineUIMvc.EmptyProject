CREATE VIEW V_TEXT1 AS
(
SELECT * FROM T_TEXT A LEFT JOIN T_DOCUMENT B ON
	A.TEXT_DOCID = B.DOC_ID LEFT JOIN T_WSJN1 C ON
	C.JN1_ID = B.DOC_NID LEFT JOIN T_WSAJ1 D ON D.WS1_ID = 
	B.DOC_AID LEFT JOIN T_MODEL E ON E.MD_ID=B.DOC_ATYPE

)
 CREATE VIEW V_WSJN AS(
 select *
 from T_WSJN1 A LEFT JOIN T_WSAJ1 B ON
	A.JN1_AJID  = B.WS1_JH LEFT JOIN T_MODEL C ON
	C.MD_ID = B.WS1_MID  LEFT JOIN T_ProjectInfo D ON D.PR_BH = 
	b.WS1_PID   )
CREATE VIEW V_XM AS
(
SELECT * FROM T_ProjectInfo A LEFT JOIN T_MODEL B ON
    A.PR_MID = B.MD_ID 
)
 CREATE VIEW V_WSAJ AS(
 select *
 from T_WSAJ1 A LEFT JOIN T_MODEL B ON
	A.WS1_MID  = B.MD_ID LEFT JOIN T_ProjectInfo C ON
	C.PR_BH = A.WS1_PID   
	  )
 CREATE VIEW V_WSJN1 AS(
 select a.*,b.*,c.*,d.*,e.MD_NAME as MainName
 from T_WSJN1 A LEFT JOIN T_WSAJ1 B ON
	A.JN1_AJID  = B.WS1_JH LEFT JOIN T_MODEL C ON
	C.MD_ID = B.WS1_MID  LEFT JOIN T_ProjectInfo D ON D.PR_BH = 
	b.WS1_PID LEFT JOIN T_MODEL e on c.MD_PID=e.MD_ID)
